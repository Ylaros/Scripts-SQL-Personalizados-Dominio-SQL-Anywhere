SELECT DISTINCT 
    bethadba.EFACUMULADOR_VIGENCIA.CODI_ACU, 
    bethadba.EFACUMULADOR.NOME_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.PARC_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CDEB_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CCRE_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CHIS_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CONTROLA_LANCAMENTOS, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_ENTRADA, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_SAIDA, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_SERVICO, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_REDUCAO_Z, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_REDUCAO_Z_BILHETE, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_CFE, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_RMD, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_BILHETE, 
    bethadba.EFACUMULADOR_VIGENCIA.LANCAR_SOMENTE_OUTROS, 
    bethadba.EFACUMULADOR_VIGENCIA.CTCP_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CHPG_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CTJR_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CHJR_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CTMT_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CHMT_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CTDT_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CHDT_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.DCTB_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA.CHCP_ACU, 
    bethadba.EFACUMULADOR_VIGENCIA_CFOP.CODI_NAT,
    bethadba.EFACUMULADOR_VIGENCIA_IMPOSTOS.CODI_IMP,
    bethadba.EFACUMULADOR_VIGENCIA.VIGENCIA_ACU 
FROM 
    bethadba.EFACUMULADOR_VIGENCIA
LEFT JOIN 
    bethadba.EFACUMULADOR_VIGENCIA_CFOP 
    ON bethadba.EFACUMULADOR_VIGENCIA_CFOP.CODI_EMP = bethadba.EFACUMULADOR_VIGENCIA.CODI_EMP 
    AND bethadba.EFACUMULADOR_VIGENCIA_CFOP.CODI_ACU = bethadba.EFACUMULADOR_VIGENCIA.CODI_ACU 
    AND bethadba.EFACUMULADOR_VIGENCIA_CFOP.VIGENCIA_ACU = bethadba.EFACUMULADOR_VIGENCIA.VIGENCIA_ACU
LEFT JOIN 
    bethadba.EFACUMULADOR_VIGENCIA_IMPOSTOS
    ON bethadba.EFACUMULADOR_VIGENCIA_IMPOSTOS.CODI_EMP = bethadba.EFACUMULADOR_VIGENCIA.CODI_EMP 
    AND bethadba.EFACUMULADOR_VIGENCIA_IMPOSTOS.CODI_ACU = bethadba.EFACUMULADOR_VIGENCIA.CODI_ACU
INNER JOIN 
    bethadba.EFACUMULADOR 
    ON bethadba.EFACUMULADOR_VIGENCIA.CODI_EMP = bethadba.EFACUMULADOR.CODI_EMP 
    AND bethadba.EFACUMULADOR_VIGENCIA.CODI_ACU = bethadba.EFACUMULADOR.CODI_ACU
WHERE 
    bethadba.EFACUMULADOR.CODI_EMP = :Empresa
